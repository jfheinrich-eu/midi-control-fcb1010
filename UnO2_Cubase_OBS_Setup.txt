/*
 * FCB1010 (UnO2 v1.4) – Cubase OBS Setup
 * Editor: UnO2 Control Center v1.5.0
 *
 * Current purpose:
 * - Footswitch 1 controls Cubase recording by sending Note 36 on MIDI channel 10.
 * - Footswitch 2 starts playback by sending Note 38 on MIDI channel 10.
 * - Footswitch 3 stops playback by sending Note 40 on MIDI channel 10.
 * - Footswitch 4 controls Cycle by sending Note 41 on MIDI channel 10.
 * - Footswitch 5 controls Click by sending Note 43 on MIDI channel 10.
 * - Footswitch 6 controls Rewind by sending Note 45 on MIDI channel 10.
 * - Footswitch 7 controls Forward by sending Note 47 on MIDI channel 10.
 * - Footswitch 8 controls Undo by sending Note 48 on MIDI channel 10.
 * - Press sends Note On with value 127.
 * - Release sends Note Off with value 0.
 *
 * Design goal:
 * - Keep the setup simple and musician-friendly.
 * - Keep one dedicated Cubase bank with clear transport navigation.
 *
 * Suggested future Cubase note layout on channel 10:
 * FS1=36, FS2=38, FS3=40, FS4=41, FS5=43,
 * FS6=45, FS7=47, FS8=48, FS9=50, FS10=52.
 */

   /* (optional – leave empty for this simple setup) */
}

{
   /* Current active Cubase effect */
   CUBASE_RECORD
   CUBASE_PLAY
   CUBASE_STOP
   CUBASE_CYCLE
   CUBASE_CLICK
   CUBASE_REWIND
   CUBASE_FORWARD
   CUBASE_UNDO
}

{
}

BANKS =

   bank 1 : CUBASE_RECORD | CUBASE_PLAY | CUBASE_STOP | CUBASE_CYCLE | CUBASE_CLICK | CUBASE_REWIND | CUBASE_FORWARD | CUBASE_UNDO |  | 
}


/* -------------------- GLOBALS -------------------- */
/* MIDI channel (10 = drums) */
CHANNEL CUBASE_MIDI_CHANNEL = 10
CHANNEL OBS_MIDI_CHANNEL = 11

VAR $off = 0

/* Documentation-only note map for future expansion
 * FS1  -> Note 36  -> Record
 * FS2  -> Note 38  -> Play
 * FS3  -> Note 40  -> Stop
 * FS4  -> Note 41  -> Cycle
 * FS5  -> Note 43  -> Click

 * FS7  -> Note 47  -> Next Marker

/* -------------------- STARTUP INIT -------------------- */
INIT_FCB =
{
   /* Keep the Cubase workflow on bank 1 for a predictable startup state */
   GotoBank bank 1
}

/* Optional per-bank init (track current bank if you need it for more advanced logic) */
INIT_BANK bank 1 =
{
   $current_bank = 1
}

EFFECT_ON CUBASE_RECORD = {
   /* Footswitch press -> start Cubase recording */
   SendMidi CUBASE_MIDI_CHANNEL NoteOn 36 $on
}

EFFECT_OFF CUBASE_RECORD = {
   /* Footswitch release -> stop Cubase recording in the MIDI Remote script */
   SendMidi CUBASE_MIDI_CHANNEL NoteOff 36 $off
}

EFFECT_ON CUBASE_PLAY = {
   /* Footswitch press -> start Cubase playback */
   SendMidi CUBASE_MIDI_CHANNEL NoteOn 38 $on
}

EFFECT_OFF CUBASE_PLAY = {
   SendMidi CUBASE_MIDI_CHANNEL NoteOff 38 $off
}

EFFECT_ON CUBASE_STOP = {
   /* Footswitch press -> stop Cubase playback */
   SendMidi CUBASE_MIDI_CHANNEL NoteOn 40 $on
}

EFFECT_OFF CUBASE_STOP = {
   SendMidi CUBASE_MIDI_CHANNEL NoteOff 40 $off
}

EFFECT_ON CUBASE_CYCLE = {
   SendMidi CUBASE_MIDI_CHANNEL NoteOn 41 $on
}

EFFECT_OFF CUBASE_CYCLE = {

}

}

EFFECT_OFF CUBASE_CLICK = {

}

   SendMidi CUBASE_MIDI_CHANNEL NoteOn 45 $on
}

EFFECT_OFF CUBASE_REWIND = {
