/*

 * Current purpose:
 * - Footswitch 1 controls Cubase recording by sending Note 36 on MIDI channel 10.
 * - Footswitch 2 starts playback by sending Note 38 on MIDI channel 10.

 * - Footswitch 4 controls Cycle by sending Note 41 on MIDI channel 10.

 * - Footswitch 8 controls Undo by sending Note 48 on MIDI channel 10.
 * - Footswitch 9 controls Click by sending Note 43 on MIDI channel 10.
 * - Press sends Note On with value 127.
 * - Release sends Note Off with value 0.
 *
 * Design goal:
 * - Keep the setup simple and musician-friendly.
 * - Keep one dedicated Cubase bank with clear transport navigation.
 *
 * Suggested future Cubase note layout on channel 10:
 * FS1=36, FS2=38, FS3=40, FS4=41, FS5=50,
 * FS6=45, FS7=47, FS8=48, FS9=43, FS10=52.

/* -------------------- STARTUP INIT -------------------- */
INIT_FCB =
{
   /* Keep the Cubase workflow on bank 1 for a predictable startup state */
   GotoBank bank 1
}

/* Optional per-bank init (track current bank if you need it for more advanced logic) */

{
   $current_bank = 1
}

EFFECT_ON CUBASE_RECORD = {
   /* Footswitch press -> start Cubase recording */
   SendMidi CUBASE_MIDI_CHANNEL NoteOn $NOTE_FS1_RECORD $NOTE_ON_VELOCITY
}

EFFECT_OFF CUBASE_RECORD = {
   /* Footswitch release -> stop Cubase recording in the MIDI Remote script */
   SendMidi CUBASE_MIDI_CHANNEL NoteOff $NOTE_FS1_RECORD $NOTE_OFF_VELOCITY
}
